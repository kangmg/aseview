ASEView.js Usage Demo

High-level JavaScript API for molecular visualization

HTML Code
<!-- 1. Load dependencies -->
<script src="three.min.js"></script>
<script src="TrackballControls.js"></script>
<script src="styles.js"></script>
<script src="aseview.js"></script>

<!-- 2. Create container -->
<div id="viewer" style="width:600px;height:400px"></div>

<!-- 3. Initialize viewer -->
<script>
const viewer = new ASEView.MolecularViewer('#viewer', {
    style: 'cartoon',
    backgroundColor: '#1f2937'
});

viewer.setData({
    symbols: ['C', 'C', 'O', 'H', 'H', 'H', 'H', 'H', 'H'],
    positions: [
        [-0.047, 0.666, 0.000],
        [-0.047, -0.866, 0.000],
        [1.204, 1.144, 0.000],
        [1.869, 0.485, 0.000],
        [-0.570, 1.035, 0.889],
        [-0.570, 1.035, -0.889],
        [0.982, -1.162, 0.000],
        [-0.557, -1.222, 0.889],
        [-0.557, -1.222, -0.889]
    ]
});

// Change style dynamically
viewer.setStyle('glossy');
</script>
Result (Ethanol - C₂H₅OH)
API: new ASEView.MolecularViewer(selector, options) - styles: cartoon, glossy, metallic, neon, bubble, rowan, 2d, grey
