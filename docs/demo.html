ASEView.js API Examples

JavaScript API for molecular visualization - copy and paste ready!

1. MolecularViewer

Simple API

Clean API for embedding static molecular structures. Automatic scene setup and bond detection.

Complete HTML Code 
<!DOCTYPE html>
<html>
<head>
    <meta charset="UTF-8">
    <title>MolecularViewer Example</title>
    <!-- CDN Dependencies -->
    <script src="https://cdnjs.cloudflare.com/ajax/libs/three.js/r128/three.min.js"></script>
    <script src="https://unpkg.com/three@0.128.0/examples/js/controls/TrackballControls.js"></script>
    <script src="https://cdn.jsdelivr.net/gh/kangmg/aseview_v2_dev@main/aseview/static/js/styles.js"></script>
    <script src="https://cdn.jsdelivr.net/gh/kangmg/aseview_v2_dev@main/aseview/static/js/aseview.js"></script>
</head>
<body>
    <div id="viewer" style="width: 600px; height: 400px;"></div>
    <script>
        // Create viewer
        const viewer = new ASEView.MolecularViewer('#viewer', {
            style: 'cartoon',
            backgroundColor: '#1f2937',
            atomSize: 0.4,
            bondThickness: 0.1
        });

        // Set molecule data (Ethanol)
        viewer.setData({
            symbols: ['C', 'C', 'O', 'H', 'H', 'H', 'H', 'H', 'H'],
            positions: [
                [-0.047, 0.666, 0.000],
                [-0.047, -0.866, 0.000],
                [1.204, 1.144, 0.000],
                [1.869, 0.485, 0.000],
                [-0.570, 1.035, 0.889],
                [-0.570, 1.035, -0.889],
                [0.982, -1.162, 0.000],
                [-0.557, -1.222, 0.889],
                [-0.557, -1.222, -0.889]
            ]
        });

        // Optional: Change style dynamically
        // viewer.setStyle('glossy');
        // viewer.setOptions({ showBond: false });
    </script>
</body>
</html>
Result (Ethanol)

2. NormalModeViewer

Simple API

Viewer for molecular vibration animations. Automatically animates atoms based on displacement vectors.

Complete HTML Code 
<!DOCTYPE html>
<html>
<head>
    <meta charset="UTF-8">
    <title>NormalModeViewer Example</title>
    <!-- CDN Dependencies -->
    <script src="https://cdnjs.cloudflare.com/ajax/libs/three.js/r128/three.min.js"></script>
    <script src="https://unpkg.com/three@0.128.0/examples/js/controls/TrackballControls.js"></script>
    <script src="https://cdn.jsdelivr.net/gh/kangmg/aseview_v2_dev@main/aseview/static/js/styles.js"></script>
    <script src="https://cdn.jsdelivr.net/gh/kangmg/aseview_v2_dev@main/aseview/static/js/aseview.js"></script>
</head>
<body>
    <div id="viewer" style="width: 600px; height: 400px;"></div>
    <script>
        // Create normal mode viewer
        const viewer = new ASEView.NormalModeViewer('#viewer', {
            style: 'cartoon',
            backgroundColor: '#1f2937',
            amplitude: 1.0,
            animationSpeed: 0.05
        });

        // Set molecule with vibration displacements (Water asymmetric stretch)
        viewer.setData({
            symbols: ['O', 'H', 'H'],
            positions: [
                [0.000, 0.000, 0.117],
                [0.000, 0.757, -0.469],
                [0.000, -0.757, -0.469]
            ],
            displacements: [
                [0.00, 0.00, -0.07],
                [0.00, 0.42, 0.56],
                [0.00, -0.42, 0.56]
            ]
        });

        // Optional: Control animation
        // viewer.pause();
        // viewer.play();
        // viewer.setAmplitude(2.0);
        // viewer.setAnimationSpeed(0.1);
    </script>
</body>
</html>
Result (Water Vibration)

3. OverlayViewer

Simple API

Compare two molecular structures by overlaying them with adjustable opacity.

Complete HTML Code 
<!DOCTYPE html>
<html>
<head>
    <meta charset="UTF-8">
    <title>OverlayViewer Example</title>
    <!-- CDN Dependencies -->
    <script src="https://cdnjs.cloudflare.com/ajax/libs/three.js/r128/three.min.js"></script>
    <script src="https://unpkg.com/three@0.128.0/examples/js/controls/TrackballControls.js"></script>
    <script src="https://cdn.jsdelivr.net/gh/kangmg/aseview_v2_dev@main/aseview/static/js/styles.js"></script>
    <script src="https://cdn.jsdelivr.net/gh/kangmg/aseview_v2_dev@main/aseview/static/js/aseview.js"></script>
</head>
<body>
    <div id="viewer" style="width: 600px; height: 400px;"></div>
    <script>
        // Create overlay viewer
        const viewer = new ASEView.OverlayViewer('#viewer', {
            style: 'cartoon',
            backgroundColor: '#1f2937',
            opacity1: 1.0,
            opacity2: 0.5
        });

        // Structure 1: Original water
        const water1 = {
            symbols: ['O', 'H', 'H'],
            positions: [
                [0.000, 0.000, 0.117],
                [0.000, 0.757, -0.469],
                [0.000, -0.757, -0.469]
            ]
        };

        // Structure 2: Stretched water (different geometry)
        const water2 = {
            symbols: ['O', 'H', 'H'],
            positions: [
                [0.000, 0.000, 0.050],
                [0.000, 0.900, -0.400],
                [0.000, -0.900, -0.400]
            ]
        };

        // Set both structures for comparison
        viewer.setStructures(water1, water2);

        // Optional: Adjust opacity
        // viewer.setOpacity(1.0, 0.3);
    </script>
</body>
</html>
Result (Water Comparison)

4. InteractiveViewer

Full UI

Complete viewer with sidebar controls. Same experience as Python viewer - style selector, toggles, and sliders.

Complete HTML Code 
<!DOCTYPE html>
<html>
<head>
    <meta charset="UTF-8">
    <title>InteractiveViewer Example</title>
    <!-- CDN Dependencies -->
    <script src="https://cdnjs.cloudflare.com/ajax/libs/three.js/r128/three.min.js"></script>
    <script src="https://unpkg.com/three@0.128.0/examples/js/controls/TrackballControls.js"></script>
    <script src="https://cdn.jsdelivr.net/gh/kangmg/aseview_v2_dev@main/aseview/static/js/styles.js"></script>
    <script src="https://cdn.jsdelivr.net/gh/kangmg/aseview_v2_dev@main/aseview/static/js/aseview.js"></script>
</head>
<body>
    <div id="viewer" style="width: 800px; height: 500px;"></div>
    <script>
        // Create interactive viewer with full UI controls
        const viewer = new ASEView.InteractiveViewer('#viewer', {
            style: 'cartoon',
            data: {
                symbols: ['C', 'C', 'O', 'H', 'H', 'H', 'H', 'H', 'H'],
                positions: [
                    [-0.047, 0.666, 0.000],
                    [-0.047, -0.866, 0.000],
                    [1.204, 1.144, 0.000],
                    [1.869, 0.485, 0.000],
                    [-0.570, 1.035, 0.889],
                    [-0.570, 1.035, -0.889],
                    [0.982, -1.162, 0.000],
                    [-0.557, -1.222, 0.889],
                    [-0.557, -1.222, -0.889]
                ]
            }
        });

        // The sidebar includes:
        // - Style selector (cartoon, glossy, metallic, neon, bubble, etc.)
        // - Bond/Cell display toggles
        // - Atom size and bond width sliders
    </script>
</body>
</html>
